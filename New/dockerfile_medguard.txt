# MedGuard 2.0 - EDITO-Compatible Docker Image
FROM python:3.12-slim

LABEL maintainer="patrickobumselu@gmail.com"
LABEL description="MedGuard 2.0 - Mediterranean Fisheries Guardian System"
LABEL version="1.0.0"

# Set working directory
WORKDIR /app

# Install system dependencies
RUN apt-get update && apt-get install -y \
    gcc \
    g++ \
    gfortran \
    libgdal-dev \
    libgeos-dev \
    libproj-dev \
    libhdf5-dev \
    libnetcdf-dev \
    git \
    curl \
    && rm -rf /var/lib/apt/lists/*

# Set environment variables
ENV PYTHONUNBUFFERED=1 \
    PYTHONDONTWRITEBYTECODE=1 \
    PIP_NO_CACHE_DIR=1 \
    DEBIAN_FRONTEND=noninteractive

# Create non-root user
RUN useradd -m -u 1000 medguard && \
    chown -R medguard:medguard /app

# Copy requirements first for better caching
COPY --chown=medguard:medguard requirements.txt .

# Install Python dependencies
RUN pip install --upgrade pip && \
    pip install -r requirements.txt

# Copy application code
COPY --chown=medguard:medguard . .

# Create necessary directories
RUN mkdir -p /app/Data /app/processed /app/models /app/logs && \
    chown -R medguard:medguard /app

# Switch to non-root user
USER medguard

# Expose Streamlit port
EXPOSE 8501

# Health check
HEALTHCHECK --interval=30s --timeout=10s --start-period=5s --retries=3 \
    CMD curl -f http://localhost:8501/_stcore/health || exit 1

# Run the dashboard
CMD ["streamlit", "run", "03_medguard_dashboard.py", \
     "--server.port=8501", \
     "--server.address=0.0.0.0", \
     "--server.headless=true", \
     "--browser.gatherUsageStats=false", \
     "--theme.base=light"]
